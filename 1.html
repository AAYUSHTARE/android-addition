import os folder_path = '/path/to/your/folder' extension = '.yxmd' # Get a list of all files in the folder files = [f for f in os.listdir(folder_path) if os.path.isfile(os.path.join(folder_path, f))] # Filter files by extension yxmd_files = [f for f in files if f.endswith(extension)] # Sort files by modification time (newest first) yxmd_files.sort(key=lambda x: os.path.getmtime(os.path.join(folder_path, x)), reverse=True) # Open the newest file if yxmd_files: newest_file = os.path.join(folder_path, yxmd_files[0]) os.system('open "{}"'.format(newest_file)) else: print("No files with the specified extension found.") import os import glob def find_newest_file(folder, keyword): search_pattern = os.path.join(folder, f"*{keyword}*") files = glob.glob(search_pattern) if files: newest_file = max(files, key=os.path.getctime) return newest_file else: return None folder_path = "/path/to/your/folder" keywords = ["Aayush", "Aditya", "akash"] for keyword in keywords: newest_file = find_newest_file(folder_path, keyword) print(f"Newest file with '{keyword}':", newest_file)
